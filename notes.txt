Add udev rules to assign aliases to usb devices
  GNU nano 6.2                rplidar.rules                          
# set the udev rule , make the device_port be fixed by rplidar
#
KERNEL=="ttyUSB*", ATTRS{idVendor}=="10c4", ATTRS{idProduct}=="ea60">
KERNEL=="ttyUSB*", ATTRS{idVendor}=="0403", ATTRS{idProduct}=="6015">
SUBSYSTEM=="sound", ATTRS{idVendor}=="0c76", ATTRS{idProduct}=="161f>
SUBSYSTEM=="sound", ATTRS{idVendor}=="0556", ATTRS{idProduct}=="0002>
KERNEL=="video*", ATTRS{idVendor}=="8086", ATTRS{idProduct}=="0b07",>
